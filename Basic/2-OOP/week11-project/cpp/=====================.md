===========================================
شرح ifstream و ofstream و fstream في C++
===========================================

╔════════════════════════════════════════════════════════════╗
║  ifstream  =  input file stream  =  للقراءة فقط          ║
╠════════════════════════════════════════════════════════════╣
║  - يفتح الملف للقراءة                                      ║
║  - لو الملف مش موجود، يفشل في الفتح                        ║
║  - العمليات: >> (استخراج), getline(), read()              ║
╚════════════════════════════════════════════════════════════╝

المثال:
    ifstream file("data.txt");
    string line;
    getline(file, line);  // نقرأ السطر


╔════════════════════════════════════════════════════════════╗
║  ofstream  =  output file stream  =  للكتابة فقط         ║
╠════════════════════════════════════════════════════════════╣
║  - يفتح الملف للكتابة                                      ║
║  - لو الملف مش موجود، ينشئه جديد                           ║
║  - لو موجود، يمحي القديم (ما لم نستخدم app)               ║
║  - العمليات: << (إدراج), write()                          ║
╚════════════════════════════════════════════════════════════╝

المثال:
    ofstream file("output.txt");
    file << "Hello World";  // نكتب في الملف


╔════════════════════════════════════════════════════════════╗
║  fstream  =  file stream  =  للقراءة والكتابة             ║
╠════════════════════════════════════════════════════════════╣
║  - يفتح الملف للقراءة والكتابة معاً                       ║
║  - لازم نحدد الوضع: in | out | app | trunc                ║
╚════════════════════════════════════════════════════════════╝

المثال:
    fstream file("data.txt", ios::in | ios::out);


===========================================
أوضاع فتح الملف (Modes)
===========================================

الوضع           | الشرح
----------------|------------------------------------------
ios::in         | للقراءة (افتراضي في ifstream)
ios::out        | للكتابة (افتراضي في ofstream)
ios::app        | الإلحاق - يضيف في نهاية الملف
ios::trunc      | المحو - يحذف المحتوى القديم
ios::binary     | وضع ثنائي (بدون معالجة للنص)


===========================================
جدول المقارنة السريع
===========================================

النوع      |    القراءة   |   الكتابة  |إنشاء ملف |  محو المحتوى
-----------|---------|---------|-----------|------------
ifstream   |   ✓    |    ✗    |     ✗     |      ✗
ofstream   |   ✗    |    ✓    |     ✓     |      ✓
fstream    |   ✓    |    ✓    |     ✓     |    حسب الوضع


===========================================
مثال: نسخ ملف
===========================================

void copy_file(string from, string to) {
    ifstream src(from);     // فتح للقراءة
    ofstream dst(to);       // فتح للكتابة
    
    string line;
    while (getline(src, line)) {
        dst << line << endl;
    }
}
